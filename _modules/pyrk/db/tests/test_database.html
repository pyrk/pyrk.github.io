[docs]    def test_add_entry(self):
        rec = {'t0': 0.0,
               'tf': 10.0,
               'dt': 0.1,
               't_feedback': 1.0,
               'iso': 'u235',
               'e': 'thermal',
               'n_pg': 1,
               'n_dg': 1,
               'kappa': 0.0,
               'plotdir': 'images'}
        tab = self.a.get_table('metadata', 'sim_info')
        self.a.add_row(tab, rec)
        t0_obs = tab.col('t0')[0]
        t0_exp = rec['t0']
        assert_equal(t0_obs, t0_exp)
        tf_obs = tab.col('tf')[0]
        tf_exp = rec['tf']
        assert_equal(tf_obs, tf_exp)
